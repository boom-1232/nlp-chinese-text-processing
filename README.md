"""

    print("=" * 50)
    print("NumPy矩阵运算示例")
    print("虚拟环境: nlp_yyr")

    print("=" * 50)

    print("\n1. 创建矩阵")

                         [7, 8, 9]])
    matrix_b = np.array([[9, 8, 7], 
                         [6, 5, 4], 
                         [3, 2, 1]])
    
    # 矩阵加法
    result_add = matrix_a + matrix_b
    
    # 矩阵乘法
    result_mul = matrix_a @ matrix_b
    
    return "矩阵运算完成！"

if __name__ == "__main__":
    result = main()
    print(f"\n{result}")
